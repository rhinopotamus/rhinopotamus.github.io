\begin{equation} \alpha = y_1 - \frac{v_1^2}{2g},\tag{3.2.2} \end{equation}

\begin{equation} \left(\frac{ds}{dt}\right)^2 = 2gy-2gy_1 + v_1^2 = 2g(y-\alpha).\tag{3.2.3} \end{equation}
